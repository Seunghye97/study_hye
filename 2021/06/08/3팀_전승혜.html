    global result1
    result1 += num
    return result1

    global result2
    result2 += num
    return result2

class Calculator:
    def __init__(self):
        self.result = 0

    def add(self, num):
        self.result += num
        return self.result
    def sub(self, num):
        self.result -= num
        return self.result

cal1 = Calculator()
cal2 = Calculator()
cal3 = Calculator()

print(cal1.add(3))
print(cal2.add(4))
print(cal3.sub(3))

class Car:
    def __init__(self,brand,model,color):
        self.brand = brand
        self.model = model
        self.color = color
        print(brand,model,color,'생성')

    def turn_on(self):
        print(f'{self.brand} {self.model} 시동을 겁니다')
    def turn_off(self):
        print(f'{self.brand} {self.model} 시동을 끕니다')
    def drive(self):
        print(f'{self.brand} {self.model} 주행중입니다')

car1 = Car('BMT','3','white')
car2 = Car('audi','A5','black')

# car1,car2 기능 수행
car1.turn_on()
car1.drive()
car1.turn_off()

car2.turn_on()
car2.drive()
car2.turn_off()

Person 클래스 만들기 

class Person:
    def __init__(self, name, age, address):
        self.hello = '안녕하세요'
        self.name = name
        self.age = age
        self.address = address

    def greeting(self):
        print(f'{self.hello} 저는 {self.name} 입니다')

p1 = Person('전승혜', 25, '서울')
p2 = Person('신효식', 30, '서울')

p1.greeting()
p2.greeting()
print('이름: ',p1.name)
print('나이: ',p1.age)
print('주소: ',p1.address)

print('이름:', p2.name)
print('나이:', p2.age)
print('주소: ',p1.address)
